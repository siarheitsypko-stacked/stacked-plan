Примитивы: number, string, boolean, null, undefined, symbol, bigint. Объекты, массивы, функции — ссылочные типы.

let age = 25;          // number
let name = "JS";       // string
let isFun = true;      // boolean
let nothing = null;    // null
let notDefined;        // undefined
let id = Symbol("id"); // symbol
let big = 9007199254740991n; // bigint

console.log(typeof big); // "bigint"

⚙️ Функции & замыкания

Function declaration, expression, стрелочные функции, параметры по умолчанию, rest, замыкания.

// declaration
function sum(a, b) { return a + b; }

// expression
const mul = function(a, b) { return a * b; };

// arrow (лексический this)
const square = x => x * x;

// параметры по умолчанию + rest
function greet(name = "Гость", ...tags) {
    return `Привет, ${name}! Теги: ${tags.join(', ')}`;
}

// замыкание
function counter() {
    let count = 0;
    return () => ++count;
}
const inc = counter();
console.log(inc()); // 1
замыканиястрелочные функцииrest parameters
🧬 Объекты и прототипы

Литералы, конструкторы, классы ES6, прототипное наследование, this, call/apply/bind.

// литерал
const user = { name: "Alice", age: 25 };

// конструктор + прототип
function Car(brand) { this.brand = brand; }
Car.prototype.drive = function() { return `${this.brand} едет`; };

// класс (синтаксический сахар)
class Animal {
    constructor(name) { this.name = name; }
    speak() { return `${this.name} издаёт звук`; }
}

// управление контекстом
function show() { console.log(this.name); }
show.call(user); // Alice
прототипное наследованиеclassthis/call/bind
📊 Массивы и деструктуризация

Функциональные методы: map, filter, reduce, find, some, every, spread/rest, деструктуризация.

const nums = [1, 2, 3, 4, 5];
const doubled = nums.map(n => n * 2); // [2,4,6,8,10]
const evens = nums.filter(n => n % 2 === 0); // [2,4]
const sum = nums.reduce((acc, n) => acc + n, 0); // 15

// деструктуризация
const [first, second] = nums;   // first=1, second=2
const { name, age } = { name: "Elena", age: 28 };
const [head, ...tail] = nums; // head=1, tail=[2,3,4,5]

⏳ Асинхронность

Promise, цепочки, async/await, обработка ошибок, Event Loop, Promise.all, Promise.race.

// Promise
const fetchData = () => {
    return fetch('https://jsonplaceholder.typicode.com/posts/1')
        .then(res => res.json())
        .then(data => console.log(data.title))
        .catch(err => console.error(err));
};

// async/await
async function getPost() {
    try {
        const res = await fetch('https://jsonplaceholder.typicode.com/posts/1');
        const data = await res.json();
        console.log(data.title);
    } catch (err) {
        console.error("Ошибка:", err);
    }
}

Promise.all([p1, p2, p3]).then(values => console.log(values));

⚠️ Обработка ошибок

try/catch/finally, создание своих ошибок, throw.

try {
    let json = '{"name":"Alex"}';
    let user = JSON.parse(json);
    if (!user.age) throw new Error("Возраст не указан");
    console.log(user.age);
} catch (error) {
    console.error("Ошибка:", error.message);
} finally {
    console.log("Очистка ресурсов");
}

class ApiError extends Error {
    constructor(status, message) {
        super(message);
        this.status = status;
        this.name = "ApiError";
    }
}
throw new ApiError(404, "Ресурс не найден");

✨ ES2020+ новинки

Опциональная цепочка ?., оператор нулевого слияния ??, BigInt, globalThis, Promise.allSettled, Array.at(), Object.groupBy (ES2024).

const user = { profile: { name: "John" } };
console.log(user?.profile?.name);    // John
console.log(user?.address?.city);    // undefined

const count = null ?? 10;            // 10
const foo = false ?? "default";      // false

const results = await Promise.allSettled([Promise.resolve(1), Promise.reject("error")]);

const arr = [10, 20, 30];
console.log(arr.at(-1)); // 30

const grouped = Object.groupBy([{age:20},{age:20},{age:30}], p => p.age);
